// PROBLEM : ARROW FUNCTION PROBLEM // IN ORDER TO UNDERSTAND THE SIMPLICITY OF ARROW FUNCTIONS LETS SORT SOME NUMARICAL ARRAYS WITH TRADITIONAL FUNCTIONS // AND THEN WITH ARROW FUNCTIONS TO SEE THE DIFFERENCE let numbers = [2,3,5,8,3,0,9,6,4,3,5,2]; // USING TRADITIONAL FUNCTIONS numbers.sort(desendingSort); numbers.sort(display); function desendingSort(a, b){ return b-a; } function display(elem){ console.log(elem); } // OK SO THIS IS DONE USING TRADITIONAL FUNCTIONS, NOW LETS GET THE SAME FUNCTIONLITY WITH LESS CODE USING ARROW FUNCTIONS // USING ARROW FUNCTIONS numbers.sort((a,b) => b-a); numbers.forEach(elem => {console.log(elem)}); // DONE, NOW LETS SEE THE DIFFERENCE THIS ARROW FUNCTION CAN BE VERY USEFULL IN SOME CASES, SO REMEMBER AND IMPLEMENT THEM IN YOUR CODE MR DANI
